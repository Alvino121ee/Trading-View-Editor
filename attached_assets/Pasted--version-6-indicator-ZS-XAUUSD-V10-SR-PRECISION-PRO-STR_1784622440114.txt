//==================== HIGH VOLUME SR FUNCTIONS ====================//
upAndDownVolume() =>
    posVol = 0.0
    negVol = 0.0
    var isBuyVolume = true

    switch
        close > open => isBuyVolume := true
        close < open => isBuyVolume := false

    if isBuyVolume
        posVol += volume
    else
        negVol -= volume

    posVol + negVol

calcSupportResistance(_src, _lookbackPeriod) =>
    Vol    = upAndDownVolume()
    vol_hi = ta.highest(Vol / 2.5, srVolLen)
    vol_lo = ta.lowest(Vol / 2.5, srVolLen)

    var float supportLevel      = na
    var float supportLevel_1    = na
    var float resistanceLevel   = na
    var float resistanceLevel_1 = na
    var box   sup               = na
    var box   res               = na
    var color res_color         = na
    var color sup_color         = na

    pivotHigh = ta.pivothigh(_src, _lookbackPeriod, _lookbackPeriod)
    pivotLow  = ta.pivotlow(_src, _lookbackPeriod, _lookbackPeriod)

    atrSR = ta.atr(200)
    widthSR = atrSR * srBoxWidth

    if not na(pivotLow) and Vol > vol_hi
        supportLevel   := pivotLow
        supportLevel_1 := supportLevel - widthSR

        if showSRBoxes
            topLeft     = chart.point.from_index(bar_index - _lookbackPeriod, supportLevel)
            bottomRight = chart.point.from_index(bar_index, supportLevel_1)
            sup_color   := color.from_gradient(Vol, 0, ta.highest(Vol, 25), color.new(color.green, 100), color.new(color.green, 30))

            sup := box.new(
                 top_left     = topLeft,
                 bottom_right = bottomRight,
                 border_color = color.green,
                 border_width = 1,
                 bgcolor      = sup_color,
                 text         = "Vol: " + str.tostring(math.round(Vol, 2)),
                 text_color   = chart.fg_color,
                 text_size    = size.small)

    if not na(pivotHigh) and Vol < vol_lo
        resistanceLevel   := pivotHigh
        resistanceLevel_1 := resistanceLevel + widthSR

        if showSRBoxes
            topLeft     = chart.point.from_index(bar_index - _lookbackPeriod, resistanceLevel)
            bottomRight = chart.point.from_index(bar_index, resistanceLevel_1)
            res_color   := color.from_gradient(Vol, ta.lowest(Vol, 25), 0, color.new(color.red, 30), color.new(color.red, 100))

            res := box.new(
                 top_left     = topLeft,
                 bottom_right = bottomRight,
                 border_color = color.red,
                 border_width = 1,
                 bgcolor      = res_color,
                 text         = "Vol: " + str.tostring(math.round(Vol, 2)),
                 text_color   = chart.fg_color,
                 text_size    = size.small)

    if not na(sup)
        box.set_right(sup, bar_index + 1)

    if not na(res)
        box.set_right(res, bar_index + 1)

    brekout_res = ta.crossover(low, resistanceLevel_1)
    res_holds   = ta.crossunder(high, resistanceLevel)

    sup_holds   = ta.crossover(low, supportLevel)
    brekout_sup = ta.crossunder(high, supportLevel_1)

    if brekout_sup and not na(sup)
        box.set_bgcolor(sup, color.new(color.red, 80))
        box.set_border_color(sup, color.red)
        box.set_border_style(sup, line.style_dashed)

    if sup_holds and not na(sup)
        box.set_bgcolor(sup, sup_color)
        box.set_border_color(sup, color.green)
        box.set_border_style(sup, line.style_solid)

    if brekout_res and not na(res)
        box.set_bgcolor(res, color.new(color.green, 80))
        box.set_border_color(res, color.new(color.green, 0))
        box.set_border_style(res, line.style_dashed)

    if res_holds and not na(res)
        box.set_bgcolor(res, res_color)
        box.set_border_color(res, color.new(color.red, 0))
        box.set_border_style(res, line.style_solid)

    [supportLevel, supportLevel_1, resistanceLevel, resistanceLevel_1, brekout_res, res_holds, sup_holds, brekout_sup]

allowSell = modeSignal == "AUTO BUY SELL" or modeSignal == "SELL ONLY"

canSignal =
     barstate.isconfirmed and
     sessionOK and
     not tradeActive and
     pendingDir == 0 and
     (na(lastExitBar) or bar_index - lastExitBar >= cooldownBars)

buySetup = (buyTrend or buyBreak or sideBuy or resRetestBuy) and not srBlockBuy and not precisionBlockBuy
sellSetup = (sellTrend or sellBreak or sideSell or supRetestSell) and not srBlockSell and not precisionBlockSell

buyValid =
     canSignal and
     allowBuy and
     buySetup and
     buyScore >= minQuality and
     buyScore >= sellScore + scoreGap

sellValid =
     canSignal and
     allowSell and
     sellSetup and
     sellScore >= minQuality and
     sellScore >= buyScore + scoreGap

buySRValid =
     canSignal and
     allowBuy and
     buySRSetup

sellSRValid =
     canSignal and
     allowSell and
     sellSRSetup

doSRBuy =
     buySRValid and
     (not sellSRValid or buyScore >= sellScore + scoreGap)

doSRSell =
     sellSRValid and
     (not buySRValid or sellScore >= buyScore + scoreGap)

// 1) BUY/SELL KUAT-SEDANG dari SR strength.

pctText(_part, _total) =>
    _total > 0 ? str.tostring(_part * 100.0 / _total, "#.##") + "%" : "-"

arrLine(_arr, _start, _count) =>
    string out = ""
    int sz = array.size(_arr)
    if sz > 0 and _start < sz
        int last = math.min(sz - 1, _start + _count - 1)
        for i = _start to last
            out += (str.length(out) > 0 ? " " : "") + array.get(_arr, i)
    out
